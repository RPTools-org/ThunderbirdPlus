def shortenUrl(lnk, label) :
	lnk = lnk.replace("https://", label)
	lnk = lnk.replace("http://", label)
	return lnk.split("/")[0]

def strBetween2(txt, sep1, sep2) :
	pos1 = txt.find(sep1) 
	if pos1 < 0 : return ""
	pos1 +=  len(sep1)
	pos2 = txt.find(sep2, pos1)
	if  pos2 < 0 : return ""
	return txt[pos1:pos2]


def getSenderName(header) :
	# On Behalf Of Isabelle Delarue via groups.ioSent 
	if "Behalf Of" in header :
		return strBetween2(header, "Behalf Of", "via groups").strip() 
	elif "&lt" in header :
		return header.split(" &lt")[0].replace('"', '').strip()
	return u"inchangé" + header.strip()
